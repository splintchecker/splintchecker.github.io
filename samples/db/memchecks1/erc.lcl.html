  modifies c;
  /* ensures c' = { }; */
}
 
void erc_insert(erc c, eref er)

  /* requires c^.activeIters = 0 /\ er \neq erefNIL; */
  modifies c;
  /* ensures c' = [insert(er, c^.val), 0]; */

| bool : void | erc_delete(erc c, eref er)

  /* ensures result = er \in c^.val

  /* ensures c1' = [c1^.val \U c2^.val, 0]; */
